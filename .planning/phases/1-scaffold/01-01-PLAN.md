      const templates = [

    **`backend-v2/src/middleware/errorHandler.ts`**:
    Global Express error handler. Returns JSON envelope `{ success: false, error: { message, code? } }`. Must have 4 parameters for Express to recognize it as error middleware.
    ```typescript
    import { Request, Response, NextFunction } from 'express'

    export class AppError extends Error {
      statusCode: number
      code?: string

      constructor(message: string, statusCode: number = 500, code?: string) {
        super(message)
        this.statusCode = statusCode
        this.code = code
        this.name = 'AppError'
      }
    }

    export function errorHandler(
      err: Error,
      _req: Request,
      res: Response,
      _next: NextFunction
    ): void {
      console.error('[Error]', err.message)

      if (err instanceof AppError) {
        res.status(err.statusCode).json({
          success: false,
          error: { message: err.message, code: err.code },
          data: null,
        })
        return
      }

      res.status(500).json({
        success: false,
        error: { message: 'Internal server error' },
        data: null,
      })
    }
    ```

    **`backend-v2/src/app.ts`**:
    Express app with middleware pipeline in correct order (security → parsing → session → routes → error handler).
    ```typescript
    import dotenv from 'dotenv'
    dotenv.config()

    import express from 'express'
    import helmet from 'helmet'
    import cors from 'cors'
    import session from 'express-session'
    import rateLimit from 'express-rate-limit'
    import routes from './routes'
    import { errorHandler } from './middleware/errorHandler'

    const app = express()

    // 1. Security headers
    app.use(helmet())

    // 2. CORS
    app.use(cors({
      origin: process.env.CORS_ORIGIN || 'http://localhost:5173',
      credentials: true,
    }))

    // 3. Body parsing
    app.use(express.json())
    app.use(express.urlencoded({ extended: true }))

    // 4. Rate limiter
    const limiter = rateLimit({
      windowMs: 15 * 60 * 1000, // 15 minutes
      max: 100,
      standardHeaders: true,
      legacyHeaders: false,
    })
    app.use('/api', limiter)

    // 5. Session
    app.use(session({
      secret: process.env.SESSION_SECRET || 'dev-secret',
      resave: false,
      saveUninitialized: false,
      cookie: {
        httpOnly: true,
        secure: false, // false for dev — set to true in production with HTTPS
        sameSite: 'lax',
        maxAge: 24 * 60 * 60 * 1000, // 24 hours
      },
    }))

    // 6. Routes
    app.use('/api', routes)

    // 7. 404 handler
    app.use((_req, res) => {
      res.status(404).json({
        success: false,
        error: { message: 'Not found' },
        data: null,
      })
    })

    // 8. Global error handler
    app.use(errorHandler)

    export { app }
    ```
