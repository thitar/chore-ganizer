<objective>
Fix three penalty performance issues: race condition in `applyOverduePenalty` (PERF-01), parallelize overdue chore processing (PERF-02), and eliminate N+1 notification queries (PERF-03).

**Purpose:** The penalty processing system has two intertwined performance problems and a race condition: 
1. `applyOverduePenalty` does a read-then-write without atomicity → double-penalty risk (PERF-01)
2. `processOverdueChores` processes chores sequentially in `for...of` (PERF-02)
3. `notifyParentOfOverdue` makes individual DB calls per parent per chore → N+1 queries (PERF-03)

**Output:** 
- Single file modification: `backend/src/services/overdue-penalty.service.ts`
- Race condition eliminated via `$transaction` atomicity
- Parallel processing with per-chore error isolation
- Batch parent notification setting fetch before the penalty loop
</objective>

      ```typescript
      export const applyOverduePenalty = async (assignmentId: number, multiplier: number) => {
        // Read + validation outside transaction (no mutation here)
        const assignment = await prisma.choreAssignment.findUnique({
          where: { id: assignmentId },
          select: { penaltyApplied: true, userId: true, choreTemplate: { select: { title: true, points: true } } }
        })
        if (!assignment) throw new AppError(`Assignment ${assignmentId} not found`, 404, 'NOT_FOUND')
        if (assignment.penaltyApplied) throw new AppError('Penalty has already been applied', 409, 'ALREADY_PENALIZED')
        
        const penaltyPoints = -Math.round(Math.abs(assignment.choreTemplate.points * multiplier))
        
        // Mutations inside transaction for atomicity
        await prisma.$transaction(async (tx) => {
          await tx.user.update({ where: { id: assignment.userId }, data: { points: { increment: penaltyPoints } } })
          await tx.choreAssignment.update({ where: { id: assignmentId }, data: { penaltyApplied: true, penaltyPoints } })
        })
        
        return { penaltyPoints, userId: assignment.userId, choreTitle: assignment.choreTemplate.title }
      }
      ```

    ```typescript
      overdueChores.map(async (assignment) => {
        try {
          const penalty = await applyOverduePenalty(assignment.id, settings.overduePenaltyMultiplier)
          const daysOverdue = calculateDaysOverdue(assignment.dueDate)
          
          // Notifications in parallel
          const notifPromises: Promise<unknown>[] = []
          
          for (const parent of parents) {
            const parentSetting = settingsMap.get(parent.id)
            notifPromises.push(
              notifyParentOfOverdue(parent.id, { 
                childName: assignment.assignedTo.name, 
                choreTitle: assignment.choreTemplate.title, 
                daysOverdue, 
                penaltyPoints: penalty.penaltyPoints 
              }, { ntfyTopic: parentSetting?.ntfyTopic ?? null })
            )
          }
          
          notifPromises.push(
            notifyChildOfPenalty(assignment.userId, { 
              choreTitle: assignment.choreTemplate.title, 
              penaltyPoints: penalty.penaltyPoints 
            })
          )
          
          await Promise.allSettled(notifPromises)
          
          return { assignmentId: assignment.id, userId: penalty.userId, choreTitle: penalty.choreTitle, penaltyPoints: penalty.penaltyPoints }
        } catch (error) {
          // Re-throw with context for outer error handling
          throw Object.assign(
            error instanceof Error ? error : new Error('Unknown error'),
            { assignmentId: assignment.id }
          )
        }
      })
    )

    // Collect results
    for (const settlement of choresWithPenalties) {
      if (settlement.status === 'fulfilled') {
        result.processed++
        result.penalties.push(settlement.value)

        const assignmentId = reason?.assignmentId ?? 'unknown'

        logger.error('Error processing assignment', { component: 'OverduePenalty', assignmentId, error: reason })
        result.errors.push({ assignmentId, error: errorMsg })
      }
    }
    ```
